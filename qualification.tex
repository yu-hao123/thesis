% Pretextual Chapters
% Patches the unnumbered chapter command \@schapter
\let\politex@original@schapter\@schapter%
\def\@schapter#1{%
	\def\temptitle{#1}%
	\def\contentstitle{\contentsname}%
	\def\bibtitle{\POLIbibliographyname}

	\ifx\temptitle\contentstitle%
		% This is "Sumario". Run original to build TOC
		% and set the ABNTaftertoc flag to true.
		\politex@original@schapter{#1}%
	\else
		% Check if this is the Bibliography
		\ifx\temptitle\bibtitle%
			% This is "References". Run original.
			% Because ABNTaftertoc is true, it will go to the
			% TOC, and Part 1 will create its bookmark.
			\politex@original@schapter{#1}%
		\else
			% This is "Resumo", "Abstract", etc.
			% 1. Manually create the PDF bookmark
			\pdfbookmark[0]{#1}{#1}%
			% 2. Run original (ABNTaftertoc is false, so no TOC entry)
			\politex@original@schapter{#1}%
		\fi
	\fi
}
\makeatother
